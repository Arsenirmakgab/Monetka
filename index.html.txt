        function toggleAdminView() {
            isAdminMode = !isAdminMode;
            const adminPanel = document.getElementById('admin-panel');
            const toggleBtn = document.getElementById('toggle-admin-btn');
            
            if (isAdminMode) {
                adminPanel.style.display = 'block';
                toggleBtn.textContent = 'Выйти из управления';
                toggleBtn.classList.add('admin-mode');
            } else {
                adminPanel.style.display = 'none';
                toggleBtn.textContent = 'Панель управления';
                toggleBtn.classList.remove('admin-mode');
            }
            renderProducts();
        }

        function renderCategories() {
            const categories = ['all', 'Электроника', 'Одежда', 'Дом', 'Детям'];
            const container = document.getElementById('categories-list');
            
            container.innerHTML = categories.map(cat => {
                const name = cat === 'all' ? 'Все товары' : cat;
                const activeClass = currentCategory === cat ? 'active' : '';
                return `<div class="category-chip ${activeClass}" onclick="changeCategory('${cat}')">${name}</div>`;
            }).join('');
        }

        function changeCategory(category) {
            currentCategory = category;
            renderCategories();
            renderProducts();
        }

        function renderProducts() {
            const container = document.getElementById('products-container');
            container.innerHTML = '';

            const filtered = currentCategory === 'all' 
                ? products 
                : products.filter(p => p.category === currentCategory);

            if (filtered.length === 0) {
                container.innerHTML = '<p style="grid-column: 1/-1; text-align: center; color: var(--text-muted); margin: 3rem 0;">В этой категории пока нет товаров.</p>';
                return;
            }

            filtered.forEach(prod => {
                const card = document.createElement('div');
                card.className = 'product-card';
                
                const mainImg = (prod.images && prod.images.length > 0 && prod.images[0] !== "") 
                    ? prod.images[0] 
                    : 'data:image/svg+xml;utf8,<svg xmlns="http://www.w3.org/2000/svg" width="100" height="100"><rect width="100%" height="100%" fill="%23ccc"/><text x="50%" y="50%" dominant-baseline="middle" text-anchor="middle">No Photo</text></svg>';

                const statusText = prod.inStock ? 'В наличии' : 'Нет в наличии';
                const statusClass = prod.inStock ? 'status-in' : 'status-out';

                let actionHtml = `<button class="card-btn" onclick="openSupportModal(${prod.id})">Написать в поддержку</button>`;
                
                if (isAdminMode) {
                    actionHtml = `
                        <div class="admin-actions">
                            <button class="admin-btn btn-edit" onclick="openProductModal(${prod.id})">Редакт.</button>
                            <button class="admin-btn btn-delete" onclick="deleteProduct(${prod.id})">Удалить</button>
                        </div>
                    `;
                }

                card.innerHTML = `
                    <span class="status-badge ${statusClass}">${statusText}</span>
                    <div class="product-img-wrapper">
                        <img src="${mainImg}" class="product-img" alt="${prod.title}">
                    </div>
                    <div class="product-info">
                        <div class="product-price">${prod.price.toLocaleString()} ₽</div>
                        <div class="product-title" title="${prod.title}">${prod.title}</div>
                        <div class="product-desc" title="${prod.desc}">${prod.desc}</div>
                        ${actionHtml}
                    </div>
                `;
                container.appendChild(card);
            });
        }

        function updateCharCount(fieldId, countId, max) {
            const val = document.getElementById(fieldId).value;
            document.getElementById(countId).textContent = `${val.length} / ${max}`;
        }

        function openSupportModal(prodId) {
            document.getElementById('support-modal').style.display = 'flex';
            if(prodId) {
                const prod = products.find(p => p.id === prodId);
                document.getElementById('sup-msg').value = `Здравствуйте! Интересует товар "${prod.title}" (ID: ${prod.id}). `;
            }
        }

        function openProductModal(prodId = null) {
            const form = document.getElementById('product-form');
            form.reset();
            document.getElementById('title-count').textContent = '0 / 50';
            document.getElementById('desc-count').textContent = '0 / 1000';

            if (prodId) {
                document.getElementById('modal-product-title').textContent = 'Редактировать товар';
                const prod = products.find(p => p.id === prodId);
                
                document.getElementById('prod-id').value = prod.id;
                document.getElementById('prod-title').value = prod.title;
                document.getElementById('prod-price').value = prod.price;
                document.getElementById('prod-category').value = prod.category;
                document.getElementById('prod-status').value = prod.inStock.toString();
                document.getElementById('prod-desc').value = prod.desc;

                const inputs = document.querySelectorAll('.prod-img-url');
                inputs.forEach((input, index) => {
                    input.value = prod.images[index] || '';
                });

                updateCharCount('prod-title', 'title-count', 50);
                updateCharCount('prod-desc', 'desc-count', 1000);
            } else {
                document.getElementById('modal-product-title').textContent = 'Добавить товар';
                document.getElementById('prod-id').value = '';
            }
            document.getElementById('product-modal').style.display = 'flex';
        }

        function closeModal(modalId) {
            document.getElementById(modalId).style.display = 'none';
        }

        function handleSupportSubmit(e) {
            e.preventDefault();
            const newMsg = {
                date: new Date().toLocaleString('ru-RU'),
                name: document.getElementById('sup-name').value,
                email: document.getElementById('sup-email').value,
                text: document.getElementById('sup-msg').value
            };
            messages.unshift(newMsg);
            localStorage.setItem('monetka_messages', JSON.stringify(messages));
            alert('Ваше сообщение успешно отправлено в поддержку!');
            closeModal('support-modal');
            renderMessages();
        }

        function handleProductSubmit(e) {
            e.preventDefault();
            const id = document.getElementById('prod-id').value;
            const title = document.getElementById('prod-title').value;
            const price = parseFloat(document.getElementById('prod-price').value);
            const category = document.getElementById('prod-category').value;
            const inStock = document.getElementById('prod-status').value === 'true';
            const desc = document.getElementById('prod-desc').value;

            const imgInputs = document.querySelectorAll('.prod-img-url');
            const images = [];
            imgInputs.forEach(input => {
                if(input.value.trim() !== '') images.push(input.value.trim());
            });

            if(images.length === 0) {
                alert('Необходимо указать хотя бы одну ссылку на фото!');
                return;
            }

            if (id) {
                const index = products.findIndex(p => p.id === parseInt(id));
                products[index] = { id: parseInt(id), title, price, category, inStock, desc, images };
            } else {
                products.push({ id: Date.now(), title, price, category, inStock, desc, images });
            }

            localStorage.setItem('monetka_products', JSON.stringify(products));
            closeModal('product-modal');
            renderProducts();
        }

        function deleteProduct(id) {
            if(confirm('Вы уверены, что хотите удалить этот товар?')) {
                products = products.filter(p => p.id !== id);
                localStorage.setItem('monetka_products', JSON.stringify(products));
                renderProducts();
            }
        }

        function renderMessages() {
            const container = document.getElementById('messages-list');
            if (messages.length === 0) {
                container.innerHTML = '<tr><td colspan="4" style="text-align:center; color: var(--text-muted)">Сообщений от пользователей пока нет.</td></tr>';
                return;
            }
            container.innerHTML = messages.map(m => `
                <tr>
                    <td style="color: var(--text-muted); font-size: 0.85rem;">${m.date}</td>
                    <td><strong>${escapeHtml(m.name)}</strong></td>
                    <td><a href="mailto:${m.email}" style="color: var(--primary-hover); text-decoration:none;">${escapeHtml(m.email)}</a></td>
                    <td>${escapeHtml(m.text)}</td>
                </tr>
            `).join('');
        }

        function escapeHtml(text) {
            const map = { '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#039;' };
            return text.replace(/[&<>"']/g, m => map[m]);
        }
